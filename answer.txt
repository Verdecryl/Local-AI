# Scriptul monitoring.sh (pentru taskul inițial)
cat << 'EOF' > monitoring.sh
#!/bin/bash

LOG_FILE="monitoring.log"

while true; do
    LOAD_AVG=$(uptime | awk -F'[ ,]+' '/load average/{print $(NF-2)}')
    MEMORY_USED=$(free -h | awk '/Mem:/{print $3}')
    DISK_USED_PERCENT=$(df -h / | tail -1 | awk '{print $5}' | sed 's/%//')
    DISK_FREE_PERCENT=$(expr 100 - "$DISK_USED_PERCENT")
    echo "$LOAD_AVG $MEMORY_USED ${DISK_FREE_PERCENT}%" >> "$LOG_FILE"
    sleep 10
done
EOF
chmod +x monitoring.sh

# b) [5p] Modificări pentru a rula scripturile de oriunde (după crearea tuturor scripturilor)
# Presupunând că scripts.sh, check_hosts.sh, user_info.sh sunt create în directorul curent.
# NOTĂ: Rulați aceasta după ce ați creat toate cele 3 scripturi.
sudo mv monitoring.sh /usr/local/bin/
sudo mv check_hosts.sh /usr/local/bin/
sudo mv user_info.sh /usr/local/bin/

# a) [5p] Script pentru verificarea hostname-urilor
mkdir -p /home/student
cat << EOF > /home/student/hosts
ocw.cs.pub.ro
badhost.org
google.com
EOF

cat << 'EOF' > check_hosts.sh
#!/bin/bash

HOSTS_FILE="/home/student/hosts"

if [ ! -f "$HOSTS_FILE" ]; then
    echo "Error: Host file $HOSTS_FILE not found."
    exit 1
fi

while IFS= read -r HOST; do
    if [ -z "$HOST" ]; then
        continue
    fi
    if ping -c 4 -W 1 "$HOST" &>/dev/null; then
        echo "$HOST HOST OK"
    else
        echo "$HOST HOST FAILED"
    fi
done < "$HOSTS_FILE"
EOF
chmod +x check_hosts.sh


# c) [5p] Script pentru afișarea UID și grupurilor userilor
cat << 'EOF' > user_info.sh
#!/bin/bash

if [ "$#" -eq 0 ]; then
    echo "Usage: $0 <user1> [user2 ...]"
    exit 1
fi

for USERNAME in "$@"; do
    if id -u "$USERNAME" &>/dev/null; then
        UID_VAL=$(id -u "$USERNAME")
        GROUPS_VAL=$(id -Gn "$USERNAME" | tr ' ' ',')
        echo "User: $USERNAME"
        echo "  UID: $UID_VAL"
        echo "  Groups: $GROUPS_VAL"
    else
        echo "User-ul $USERNAME nu există."
    fi
done
EOF
chmod +x user_info.sh
